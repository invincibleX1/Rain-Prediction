function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("ESP Data");
    var data = JSON.parse(e.postData.contents);
    
    // Log incoming data (optional)
    Logger.log("Received: " + JSON.stringify(data));

    sheet.appendRow([new Date(), data.pressure, data.temperature, data.humidity]);
    return ContentService.createTextOutput("Success");
  } catch (err) {
    Logger.log("Error: " + err);
    return ContentService.createTextOutput("Error");
  }
}
